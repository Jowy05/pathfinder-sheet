order, turn indicator, round counter, add/delete rows, DEX tiebreaker, is-player
flag (gold / crimson styling), HP cur/max, notes, and persistence via
`exportData()` / `importData()` (the file's `collectCharacter` / `applyCharacter`
equivalents) under the key `__combat_tracker`.

Integration: the existing `rollInitiative()` button (line 231) is wired so that
when rolled, the PC is auto-added/updated in the combatant list using the
`char-name` value and the current initiative bonus reading from
`initiative-total`.

```js

/* ═══════════════════════════════════════════════════════════════
   P-7 · INITIATIVE TRACKER / TURN ORDER
   ═══════════════════════════════════════════════════════════════ */
let _combatState = { round: 1, currentIdx: 0, combatants: [] };

function iniMakeId() {
  return 'c_' + Date.now().toString(36) + '_' + Math.floor(Math.random() * 1e6).toString(36);
}

function iniAddCombatant(preset) {
  const c = Object.assign({
    id:      iniMakeId(),
    name:    '',
    init:    0,
    dex:     0,
    player:  false,
    hpCur:   '',
    hpMax:   '',
    notes:   '',
  }, preset || {});
  // Avoid duplicates when auto-adding the PC by id or name
  if (preset && preset.player && preset.name) {
    const exist = _combatState.combatants.find(x => x.player && x.name === preset.name);
    if (exist) {
      if (preset.init  !== undefined) exist.init  = preset.init;
      if (preset.dex   !== undefined) exist.dex   = preset.dex;
      if (preset.hpCur !== undefined) exist.hpCur = preset.hpCur;
      if (preset.hpMax !== undefined) exist.hpMax = preset.hpMax;
      iniRender();
      return exist;
    }
  }
  _combatState.combatants.push(c);
  iniRender();
  return c;
}

function iniRemoveCombatant(id) {
  const i = _combatState.combatants.findIndex(x => x.id === id);
  if (i < 0) return;
  _combatState.combatants.splice(i, 1);
  if (_combatState.currentIdx >= _combatState.combatants.length) {
    _combatState.currentIdx = 0;
  }
  iniRender();
}

function iniUpdateField(id, field, value) {
  const c = _combatState.combatants.find(x => x.id === id);
  if (!c) return;
  if (field === 'init' || field === 'dex') {
    c[field] = parseInt(value, 10) || 0;
  } else if (field === 'player') {
    c.player = !!value;
  } else {
    c[field] = value;
  }
  // Update row class + turn label without full re-render for text typing
  const row = document.querySelector(`#ini-tbody tr[data-id="${id}"]`);
  if (row) {
    row.classList.toggle('is-player', !!c.player);
    row.classList.toggle('is-enemy',  !c.player);
  }
  if (field === 'name') iniUpdateTurnLabel();
}

function iniSort() {
  // Preserve the currently active combatant reference
  const activeId = _combatState.combatants[_combatState.currentIdx]?.id;
  _combatState.combatants.sort((a, b) => {
    if (b.init !== a.init) return b.init - a.init;
    return (b.dex || 0) - (a.dex || 0);
  });
  if (activeId) {
    const ni = _combatState.combatants.findIndex(x => x.id === activeId);
    if (ni >= 0) _combatState.currentIdx = ni;
  }
  iniRender();
}

function iniNextTurn() {
  if (!_combatState.combatants.length) return;
  _combatState.currentIdx++;
  if (_combatState.currentIdx >= _combatState.combatants.length) {
    _combatState.currentIdx = 0;
    _combatState.round++;
  }
  iniRender();
}

function iniReset() {
  const msg = (typeof t === 'function') ? t('m.iniResetConfirm') : '¿Reiniciar el combate? Se borrarán todos los combatientes.';
  if (!confirm(msg)) return;
  _combatState = { round: 1, currentIdx: 0, combatants: [] };
  iniRender();
}

function toggleIniTracker() {
  const p = document.getElementById('initiative-tracker-panel');
  if (!p) return;
  p.classList.toggle('collapsed');
  const btn = p.querySelector('.ini-collapse-btn');
  if (btn) btn.textContent = p.classList.contains('collapsed') ? '▸' : '▾';
}

function iniUpdateTurnLabel() {
  const lbl = document.getElementById('ini-turn-name');
  if (!lbl) return;
  const c = _combatState.combatants[_combatState.currentIdx];
  lbl.textContent = c ? (c.name || '—') : '—';
}

function iniRender() {
  const tbody = document.getElementById('ini-tbody');
  const empty = document.getElementById('ini-empty');
  const roundEl = document.getElementById('ini-round-num');
  if (!tbody || !empty || !roundEl) return;
  roundEl.textContent = _combatState.round;
  tbody.innerHTML = '';
  if (!_combatState.combatants.length) {
    empty.style.display = '';
    iniUpdateTurnLabel();
    return;
  }
  empty.style.display = 'none';
  _combatState.combatants.forEach((c, idx) => {
    const tr = document.createElement('tr');
    tr.dataset.id = c.id;
    tr.className = 'ini-row ' + (c.player ? 'is-player' : 'is-enemy') +
                   (idx === _combatState.currentIdx ? ' is-active' : '');
    tr.innerHTML =
      '<td><span class="ini-arrow"></span></td>' +
      '<td><input type="text" class="ini-name" value="' + _iniEsc(c.name) + '"' +
           ' oninput="iniUpdateField(\'' + c.id + '\',\'name\',this.value)"></td>' +
      '<td><input type="number" class="ini-init" value="' + (c.init|0) + '"' +
           ' oninput="iniUpdateField(\'' + c.id + '\',\'init\',this.value)"></td>' +
      '<td><input type="number" class="ini-dex" value="' + (c.dex|0) + '"' +
           ' oninput="iniUpdateField(\'' + c.id + '\',\'dex\',this.value)"></td>' +
      '<td><label class="ini-player-label">' +
        '<input type="checkbox"' + (c.player ? ' checked' : '') +
        ' onchange="iniUpdateField(\'' + c.id + '\',\'player\',this.checked)"> PJ</label></td>' +
      '<td style="white-space:nowrap;">' +
        '<input type="number" class="ini-hp" value="' + (c.hpCur === '' ? '' : (c.hpCur|0)) + '"' +
          ' oninput="iniUpdateField(\'' + c.id + '\',\'hpCur\',this.value)"> / ' +
        '<input type="number" class="ini-hp" value="' + (c.hpMax === '' ? '' : (c.hpMax|0)) + '"' +
          ' oninput="iniUpdateField(\'' + c.id + '\',\'hpMax\',this.value)"></td>' +
      '<td><input type="text" class="ini-notes" value="' + _iniEsc(c.notes) + '"' +
           ' oninput="iniUpdateField(\'' + c.id + '\',\'notes\',this.value)"></td>' +
      '<td><button type="button" class="ini-del" title="Eliminar"' +
           ' onclick="iniRemoveCombatant(\'' + c.id + '\')">✕</button></td>';
    tbody.appendChild(tr);
  });
  iniUpdateTurnLabel();
}

function _iniEsc(s) {
  return String(s == null ? '' : s)
    .replace(/&/g, '&amp;').replace(/"/g, '&quot;')
    .replace(/</g, '&lt;').replace(/>/g, '&gt;');
}

// Auto-add the PC when initiative is rolled (optional integration).
function iniAutoAddPC() {
  const name = (document.getElementById('char-name')?.value || '').trim();
  if (!name) return;
  const initTotalEl = document.getElementById('initiative-total');
  let bonus = 0;
  if (initTotalEl) {
    const m = (initTotalEl.value || '').match(/-?\d+/);
    if (m) bonus = parseInt(m[0], 10) || 0;
  }
  const dexMod = (typeof getAttrMod === 'function') ? (getAttrMod('dex') || 0) : 0;
  const hpCur = parseInt(document.getElementById('hp-current')?.value, 10);
  const hpMax = parseInt(document.getElementById('hp-max')?.value, 10);
  iniAddCombatant({
    name,
    init: bonus,
    dex:  dexMod,
    player: true,
    hpCur: isNaN(hpCur) ? '' : hpCur,
    hpMax: isNaN(hpMax) ? '' : hpMax,
  });
}
```

## 6. Notes

### 6.1 · Placement rationale
- The panel lives on its own full-width row between `#conditions-panel` and
  `#skills-panel` so the per-character combat panels keep their narrow column
  layout and the tracker gets the full width it needs for the table.
- Inserting as `panel full-row` reuses the existing `.panel` / `.panel h2`
  aesthetics automatically (leather borders, gold underline, crimson H2).

### 6.2 · Color coding
- Players: gold tint `rgba(200,160,96,0.10)` + `var(--gold-dk)` left border.
- Enemies: crimson tint `rgba(122,0,16,0.08)` + `var(--crimson)` left border.
- Active row: gold inset ring + brighter gradient, plus `▶` arrow glyph in
  crimson inside the first cell.
- All colors use existing CSS variables (`--gold`, `--gold-dk`, `--crimson`,
  `--leather`, `--leather-dk`, `--parchment`, `--parchment-dk`, `--ink`,
  `--ff-head`, `--ff-body`, `--input-bg`).

### 6.3 · Persistence model
- Stored under a single JSON-serialized key `data['__combat_tracker']` matching
  the convention used for `__notes_data`, `__session_log`, `__comp_skill_ranks`.
- Shape: `{ round:number, currentIdx:number, combatants:[{id,name,init,dex,player,hpCur,hpMax,notes}] }`.
- localStorage persistence is handled transitively by export/import; if the
  project later adds a true `saveToStorage()` pair, the same key can be reused.
  (The task spec mentions `collectCharacter`/`applyCharacter` but the actual
  functions in this file are `exportData`/`importData`; this patch hooks into
  those, which is the only persistence pipeline present.)

### 6.4 · Integration with `rollInitiative`
- The existing button at line ~231 is untouched; no markup edit needed.
- `rollInitiative()` now calls `iniAutoAddPC()` at the top. This upserts the
  PC by name into the tracker, updating init / DEX tiebreaker / HP every time
  the button is clicked. Enemies stay manual.
- If `char-name` is empty, the auto-add is skipped silently (no noisy row).

### 6.5 · Sort stability
- `iniSort()` captures the active combatant's `id` before sorting and restores
  `currentIdx` after, so sorting mid-combat does not break the turn pointer.

### 6.6 · Round counter rules
- Advances by one whenever `iniNextTurn()` loops the pointer back to index 0.
- `iniReset()` resets both round and pointer; it requires confirm(), i18n'd
  under `m.iniResetConfirm`.

### 6.7 · Collapsible
- The h2 hosts a ▾ / ▸ toggle (`toggleIniTracker`) that toggles the `collapsed`
  class on the panel; only the `.ini-body` inside is hidden so the header and
  round badge can be shown if desired later — currently both are hidden as
  well because `.ini-body` wraps everything below the h2.

### 6.8 · Print
- Buttons and the delete control are hidden on print via the existing print
  media-query scheme.
