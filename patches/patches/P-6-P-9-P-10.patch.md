## 3. JS — new helpers

Anchor: `// #ANCHOR:P6-DAMAGE-TYPE-JS-HELPERS-V1`

Place this block immediately AFTER the `getActiveTraitIds` helper at ~line 5136-5139 (sibling utility zone):

```js
// #ANCHOR:P6-DAMAGE-TYPE-JS-HELPERS-V1
const DAMAGE_TYPES = [
  { id:'piercing'    },
  { id:'slashing'    },
  { id:'bludgeoning' },
  { id:'fire'        },
  { id:'cold'        },
  { id:'acid'        },
  { id:'electricity' },
  { id:'sonic'       },
  { id:'force'       },
  { id:'negative'    },
  { id:'positive'    },
];
function _dtypeLabel(id) { return t('dtype.'+id); }

function buildDamageTypeChips(activeArr) {
  const active = new Set(activeArr || []);
  return DAMAGE_TYPES.map(dt => {
    const cls = active.has(dt.id) ? 'dtype-chip active' : 'dtype-chip';
    return `<span class="${cls}" data-dtype="${dt.id}" title="${_dtypeLabel(dt.id)}" onclick="toggleDamageTypeChip(this)">${_dtypeLabel(dt.id)}</span>`;
  }).join('');
}

function toggleDamageTypeChip(chip) {
  chip.classList.toggle('active');
  const card = chip.closest('.weapon-card');
  if (!card) return;
  // Sync card dataset
  const ids = Array.from(card.querySelectorAll('.weapon-dtype-row .dtype-chip.active'))
    .map(c => c.dataset.dtype);
  if (ids.length) card.dataset.damageTypes = ids.join(',');
  else delete card.dataset.damageTypes;
  // Refresh display so the mini-chips next to DAÑO TOTAL update
  if (typeof recalc === 'function') recalc();
}

function getWeaponDamageTypes(card) {
  return Array.from(card.querySelectorAll('.weapon-dtype-row .dtype-chip.active'))
    .map(c => c.dataset.dtype);
}

function buildDmgTypeMiniChips(ids) {
  if (!ids || !ids.length) return '';
  const html = ids.map(id =>
    `<span class="dtype-minichip" data-dtype="${id}" title="${_dtypeLabel(id)}">${_dtypeLabel(id)}</span>`
  ).join('');
  return `<span class="weapon-dmg-typechips">${html}</span>`;
}

// Maps DB.equipment.weapons `type` string ("P","S","B","P/S"...) to chip ids.
function seedDamageTypesFromDB(weaponData) {
  const t_ = (weaponData && weaponData.type ? String(weaponData.type) : '').toUpperCase();
  const out = [];
  if (t_.includes('P')) out.push('piercing');
  if (t_.includes('S')) out.push('slashing');
  if (t_.includes('B')) out.push('bludgeoning');
  return out;
}
```

```js
// #ANCHOR:P9-SLA-AT-WILL-JS-MARKER-V1

function addSlaAtWill(name='', cl='', dc='', desc='') {
  const container = document.getElementById('sla-atwill-container');
  if (!container) return;
  const div = document.createElement('div');
  div.className = 'sla-card';
  div.innerHTML = `
    <input type="text" class="sla-name"  placeholder="${t('sla.ph.name')}" value="${_escAttr(name)}">
    <input type="text" class="sla-cl"    placeholder="${t('sla.ph.cl')}"   value="${_escAttr(cl)}"  title="${t('sla.tt.cl')}">
    <input type="text" class="sla-dc"    placeholder="${t('sla.ph.dc')}"   value="${_escAttr(dc)}"  title="${t('sla.tt.dc')}">
    <input type="text" class="sla-desc"  placeholder="${t('sla.ph.desc')}" value="${_escAttr(desc)}">
    <button type="button" class="btn-sla-roll" onclick="rollSlaAtWill(this)"
            title="${t('sla.tt.roll')}">${t('btn.slaRoll')}</button>
    <button type="button" class="sla-remove" onclick="this.closest('.sla-card').remove()"
            aria-label="${t('aria.delete')}" title="${t('aria.delete')}">×</button>`;
  container.appendChild(div);
}

function rollSlaAtWill(btn) {
  const card = btn.closest('.sla-card');
  if (!card) return;
  const name = card.querySelector('.sla-name').value.trim() || t('sla.unnamed');
  const cl   = card.querySelector('.sla-cl').value.trim();
  const dc   = card.querySelector('.sla-dc').value.trim();
  const desc = card.querySelector('.sla-desc').value.trim();

  const parts = [];
  if (cl)   parts.push(`${t('sla.lbl.cl')} ${cl}`);
  if (dc)   parts.push(`${t('sla.lbl.dc')} ${dc}`);
  if (desc) parts.push(desc);

  const time = new Date().toLocaleTimeString(LANG === 'en' ? 'en-US' : 'es-ES',
                 { hour:'2-digit', minute:'2-digit' });
  const msg  = t('sla.msg.trigger', { name });
  _appendSlaLog(time, name, msg, parts.join(' · '));
}

function _appendSlaLog(time, name, msg, extra) {
  const log = document.getElementById('sla-log');
  if (!log) return;
  const line = document.createElement('div');
  line.className = 'sla-log-line';
  line.innerHTML =
    `<span class="sla-log-time">[${time}]</span>` +
    `<span class="sla-log-name">${_escHtml(name)}</span> — ` +
    `${_escHtml(msg)}` +
    (extra ? ` <em style="color:var(--leather-lt);">(${_escHtml(extra)})</em>` : '');
  log.appendChild(line);
  log.scrollTop = log.scrollHeight;
}

// Small utils (only add if not already present elsewhere)
function _escHtml(s){ return String(s).replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c])); }
function _escAttr(s){ return _escHtml(s); }
```

> If `_escHtml` already exists globally, omit the duplicate; the grep check is `grep -n "function _escHtml"` — only add if 0 matches.

## 3. JS — new helpers

Anchor: `// #ANCHOR:P10-WEAKNESS-VULNERABILITY-JS-MARKER-V1`

```js
// #ANCHOR:P10-WEAKNESS-VULNERABILITY-JS-MARKER-V1
let _weaknesses = []; // [{type, note}]

function _wkLabel(typeId) {
  // Damage-type ids fall back to dtype.*, special ones use wk.*
  const dtypeKeys = ['piercing','slashing','bludgeoning','fire','cold','acid',
                     'electricity','sonic','force','negative','positive'];
  if (dtypeKeys.includes(typeId)) return t('dtype.'+typeId);
  if (typeId === 'silver')   return t('wk.silver');
  if (typeId === 'coldiron') return t('wk.coldiron');
  if (typeId === 'sun')      return t('wk.sun');
  if (typeId === 'holy')     return t('wk.holy');
  if (typeId === 'custom')   return t('wk.custom');
  return typeId;
}

function addWeakness(type, note) {
  const typeEl = document.getElementById('wk-type-new');
  const noteEl = document.getElementById('wk-note-new');
  if (type == null) type = typeEl ? typeEl.value : '';
  if (note == null) note = noteEl ? noteEl.value.trim() : '';
  if (!type) return;
  _weaknesses.push({ type, note });
  if (noteEl) noteEl.value = '';
  renderWeaknesses();
}

function removeWeakness(idx) {
  _weaknesses.splice(idx, 1);
  renderWeaknesses();
}

function renderWeaknesses() {
  const box = document.getElementById('weaknesses-container');
  const hid = document.getElementById('weaknesses-json');
  if (!box) return;
  box.innerHTML = _weaknesses.map((w, i) => {
    const label = _wkLabel(w.type);
    const noteHtml = w.note ? `<span class="wk-note">${_escHtml(w.note)}</span>` : '';
    return `<span class="wk-chip" data-type="${w.type}">
      <span class="wk-x2">×2</span>
      <span class="wk-label">${_escHtml(label)}</span>
      ${noteHtml}
      <button type="button" class="wk-del" onclick="removeWeakness(${i})"
              title="${t('aria.delete')}" aria-label="${t('aria.delete')}">×</button>
    </span>`;
  }).join('');
  if (hid) hid.value = JSON.stringify(_weaknesses);
}
```

## 6. Notes

- **Shared taxonomy with P-6**: weakness `type` values match `dtype.*` ids where applicable (`fire`, `cold`, `electricity`, …), so a future "incoming damage auto-resolver" can match a weapon's `damageTypes` chip list (P-6) against `_weaknesses` (P-10) to detect ×2 situations.
- **Custom types**: selecting `custom` uses the label literally; users can type the true type in the note field (e.g. type=`custom`, note=`"agua bendita"`).
- **Multiple entries of the same type**: allowed (e.g. `{fire, "luz solar directa"}` and `{fire, "dragones"}`) — we do not deduplicate.
- **UI placement justification**: the weaknesses block sits between ER icons and Visión Especial, keeping all damage-interaction data contiguous within the defensas especiales column.
- **Language switching**: chips re-render via `renderWeaknesses()` which re-invokes `t()`. Hook into the existing language-change handler (grep for `applyI18N(`) by adding `renderWeaknesses();` at its end.
- **No tracker, no recalc hook**: intentionally passive; this avoids perf cost on every `recalc()` (called on most input events).

## Rollback / guard-rails

Each patch block is gated by its `#ANCHOR:…-V1` comment marker. A rollback script can `sed`-delete the range between `#ANCHOR:Pn-…-V1` and the next function boundary to undo a single patch cleanly. No existing function body is altered destructively — all changes are additive except for the two insertions inside `addWeapon()` (template append) and the single insertion inside the `recalc()` weapon loop, both of which are purely additive `setText`/`innerHTML` decorations that fail silently if the new DOM nodes are absent.
