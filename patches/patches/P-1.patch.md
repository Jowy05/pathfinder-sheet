# P-1 · Buff Tracker · Patch Spec

## Context / Naming notes

This sheet does NOT use functions literally named `collectCharacter` / `applyCharacter`. The
equivalent persistence entry points are:

- **`exportData()`** — around line **6505** of `index.html` (acts as `collectCharacter`; writes to a
  downloadable JSON blob).
- **`importData(event)`** — around line **6586** of `index.html` (acts as `applyCharacter`; reads a
  JSON file into DOM state).

There is no localStorage-based character autosave in this SPA — persistence is export/import JSON.
Accordingly, buff-tracker state is stored inside the existing JSON payload under a namespaced key
(`__buff_tracker`), matching the `__weapons_html`, `__notes_data`, `__session_log` convention used
throughout.

There is also a pre-existing, unrelated "Buffs Mágicos" concept (`buffs-container` +
`addBuff(...)`) that computes `gatherBuffDeltas()` for stat recalc. That system is preserved
untouched. P-1 adds a **separate, narrative-focused Buff Tracker** that tracks name/type/bonus/
target/duration of active magical/combat buffs for bookkeeping — it does NOT feed into `recalc()`.
(The existing `.buff-card` class is reused by `gatherBuffDeltas`, so P-1 uses a distinct class
prefix `btr-` / `buff-tracker-` to avoid collision.)

```js

/* ═══════════════════════════════════════════════════════════════
   P-1 · BUFF TRACKER
   Independent from gatherBuffDeltas(); persisted inside exportData
   under __buff_tracker. Does not feed into recalc().
   ═══════════════════════════════════════════════════════════════ */
const BTR_TYPES = [
  'moral','resistance','sacred','profane','luck','dodge','insight',
  'competence','enhancement','alchemical','size','circumstance','untyped'
];
const BTR_UNITS = ['rd','min','h','permanent'];
let _btrMinTick = 0;   // counts "Siguiente ronda" clicks toward the next minute tick (0–9)

function _btrTypeOptions(sel) {
  return BTR_TYPES.map(k =>
    `<option value="${k}"${k===sel?' selected':''}>${t('btr.type.'+k)}</option>`
  ).join('');
}
function _btrUnitOptions(sel) {
  return BTR_UNITS.map(k =>
    `<option value="${k}"${k===sel?' selected':''}>${t('btr.unit.'+k)}</option>`
  ).join('');
}
function _btrSignedFormat(n) {
  const v = parseInt(n) || 0;
  return v >= 0 ? `+${v}` : `${v}`;
}
function _btrRemainingDisplay(row) {
  const unit = row.querySelector('.btr-unit')?.value || 'rd';
  const rem  = parseInt(row.dataset.remaining);
  if (unit === 'permanent') return '∞';
  if (!isFinite(rem))       return '—';
  return String(rem);
}
function _btrRecomputeRow(row) {
  const unit = row.querySelector('.btr-unit')?.value || 'rd';
  const rem  = parseInt(row.dataset.remaining);
  const expired = (unit !== 'permanent') && isFinite(rem) && rem <= 0;
  row.classList.toggle('btr-expired', !!expired);
  const remEl = row.querySelector('.btr-remaining');
  if (remEl) remEl.value = _btrRemainingDisplay(row);
  // keep type stripe in sync
  const typeSel = row.querySelector('.btr-type');
  if (typeSel) row.dataset.btype = typeSel.value;
}
function _btrUpdateCounts() {
  const list = document.getElementById('buff-tracker-list');
  if (!list) return;
  const rows = list.querySelectorAll('.buff-tracker-row');
  let act = 0, exp = 0;
  rows.forEach(r => r.classList.contains('btr-expired') ? exp++ : act++);
  const a = document.getElementById('btr-count-active');
  const e = document.getElementById('btr-count-expired');
  const m = document.getElementById('btr-min-pending');
  if (a) a.textContent = act;
  if (e) e.textContent = exp;
  if (m) m.textContent = `${_btrMinTick}/10`;
}
function btrAddBuff(data) {
  data = data || {};
  const list = document.getElementById('buff-tracker-list');
  if (!list) return;
  const row = document.createElement('div');
  row.className = 'buff-tracker-row';
  const durVal  = (data.duration  != null) ? data.duration  : 10;
  const durUnit = data.unit || 'rd';
  const remaining = (data.remaining != null)
    ? data.remaining
    : (durUnit === 'permanent' ? Infinity : parseInt(durVal) || 0);
  row.dataset.remaining = (remaining === Infinity) ? 'Infinity' : remaining;
  row.dataset.btype     = data.btype || 'untyped';
  row.innerHTML = `
    <input type="text" class="btr-name"   value="${(data.name||'').replace(/"/g,'&quot;')}"
           data-i18n-ph="btr.ph.name" placeholder="Nombre del buff">
    <select class="btr-type">${_btrTypeOptions(data.btype || 'untyped')}</select>
    <input type="number" class="btr-bonus" value="${parseInt(data.bonus)||0}" step="1"
           data-i18n-title="btr.tt.bonus" title="Bonificador (ej: +2, -1)">
    <input type="text" class="btr-target" value="${(data.target||'').replace(/"/g,'&quot;')}"
           data-i18n-ph="btr.ph.target" placeholder="atq, CA, Fort…">
    <input type="number" class="btr-duration" min="0" value="${parseInt(durVal)||0}"
           data-i18n-title="btr.tt.duration" title="Duración">
    <select class="btr-unit">${_btrUnitOptions(durUnit)}</select>
    <input type="text" class="btr-remaining" readonly
           data-i18n-title="btr.tt.remaining" title="Rondas/unidades restantes">
    <button type="button" class="btr-del" aria-label="Eliminar" data-i18n-aria="btr.del"
            data-i18n-title="btr.del" title="Eliminar">×</button>
  `;
  list.appendChild(row);

  // listeners
  row.querySelector('.btr-duration').addEventListener('input', e => {
    const u = row.querySelector('.btr-unit').value;
    row.dataset.remaining = (u === 'permanent') ? 'Infinity' : (parseInt(e.target.value)||0);
    _btrRecomputeRow(row); _btrUpdateCounts();
  });
  row.querySelector('.btr-unit').addEventListener('change', () => {
    const u = row.querySelector('.btr-unit').value;
    const d = parseInt(row.querySelector('.btr-duration').value) || 0;
    row.dataset.remaining = (u === 'permanent') ? 'Infinity' : d;
    _btrRecomputeRow(row); _btrUpdateCounts();
  });
  row.querySelector('.btr-type').addEventListener('change', () => _btrRecomputeRow(row));
  row.querySelector('.btr-del').addEventListener('click', () => {
    row.remove(); _btrUpdateCounts();
  });

  _btrRecomputeRow(row);
  _btrUpdateCounts();
  if (typeof applyI18n === 'function') applyI18n(row);
}
function btrNextRound() {
  _btrMinTick = (_btrMinTick + 1) % 10;
  const tickMinutes = (_btrMinTick === 0); // every 10th click fires a minute tick
  document.querySelectorAll('#buff-tracker-list .buff-tracker-row').forEach(row => {
    if (row.classList.contains('btr-expired')) return;
    const unit = row.querySelector('.btr-unit').value;
    if (unit === 'permanent' || unit === 'h') return; // hours only decrement on rest
    let rem = parseInt(row.dataset.remaining);
    if (!isFinite(rem)) return;
    if (unit === 'rd')  rem = rem - 1;
    if (unit === 'min' && tickMinutes) rem = rem - 1;
    row.dataset.remaining = rem;
    _btrRecomputeRow(row);
  });
  _btrUpdateCounts();
}
function btrClearExpired() {
  document.querySelectorAll('#buff-tracker-list .buff-tracker-row.btr-expired')
    .forEach(r => r.remove());
  _btrUpdateCounts();
}
function btrSerialize() {
  const out = [];
  document.querySelectorAll('#buff-tracker-list .buff-tracker-row').forEach(row => {
    const remRaw = row.dataset.remaining;
    out.push({
      name:      row.querySelector('.btr-name')?.value      || '',
      btype:     row.querySelector('.btr-type')?.value      || 'untyped',
      bonus:     parseInt(row.querySelector('.btr-bonus')?.value) || 0,
      target:    row.querySelector('.btr-target')?.value    || '',
      duration:  parseInt(row.querySelector('.btr-duration')?.value) || 0,
      unit:      row.querySelector('.btr-unit')?.value      || 'rd',
      remaining: (remRaw === 'Infinity') ? null : (parseInt(remRaw) || 0),
      expired:   row.classList.contains('btr-expired'),
    });
  });
  return { buffs: out, minTick: _btrMinTick };
}
function btrRestore(payload) {
  const list = document.getElementById('buff-tracker-list');
  if (!list) return;
  list.innerHTML = '';
  _btrMinTick = 0;
  if (!payload) { _btrUpdateCounts(); return; }
  if (Array.isArray(payload)) payload = { buffs: payload, minTick: 0 };
  _btrMinTick = Math.max(0, Math.min(9, parseInt(payload.minTick)||0));
  (payload.buffs || []).forEach(b => {
    btrAddBuff({
      name: b.name, btype: b.btype, bonus: b.bonus, target: b.target,
      duration: b.duration, unit: b.unit,
      remaining: (b.unit === 'permanent') ? Infinity
                 : (b.remaining != null ? b.remaining : b.duration),
    });
  });
  _btrUpdateCounts();
}
document.addEventListener('DOMContentLoaded', () => {
  const add  = document.getElementById('btr-add-btn');
  const next = document.getElementById('btr-next-round-btn');
  const clr  = document.getElementById('btr-clear-btn');
  if (add)  add.addEventListener('click',  () => btrAddBuff());
  if (next) next.addEventListener('click', () => btrNextRound());
  if (clr)  clr.addEventListener('click',  () => btrClearExpired());
  _btrUpdateCounts();
});
```
